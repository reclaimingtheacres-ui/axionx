2️⃣ Schedule Model
class Schedule(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    job_id = db.Column(db.Integer, db.ForeignKey("job.id"), nullable=False)
    booking_type_id = db.Column(db.Integer, db.ForeignKey("booking_type.id"), nullable=False)
    scheduled_for = db.Column(db.DateTime, nullable=False)
    status = db.Column(db.String(50), default="Booked")  # Booked / Completed / Cancelled
    notes = db.Column(db.Text)
    created_by = db.Column(db.Integer)
    created_at = db.Column(db.DateTime, default=datetime.utcnow)

5️⃣ Flask Routes
Add Booking
@bp.post("/jobs/<int:job_id>/schedule")
def add_schedule(job_id):
    job = Job.query.get_or_404(job_id)

    date_str = request.form["schedule_date"]
    time_str = request.form["schedule_time"]
    booking_type_id = request.form["booking_type_id"]
    notes = request.form.get("notes")

    scheduled_for = datetime.strptime(
        f"{date_str} {time_str}".upper().strip(),
        "%d/%m/%Y %I:%M %p"
    )

    schedule = Schedule(
        job_id=job.id,
        booking_type_id=booking_type_id,
        scheduled_for=scheduled_for,
        notes=notes,
        created_by=current_user.id
    )

    db.session.add(schedule)
    db.session.commit()

    return redirect(url_for("jobs.view", job_id=job.id))
Add Booking Type
@bp.post("/booking-type")
def add_booking_type():
    name = request.form["new_booking_type"].strip()

    if name:
        bt = BookingType(name=name)
        db.session.add(bt)
        db.session.commit()

    return redirect(request.referrer)
