🔧 2. ADD THESE HELPER FUNCTIONS (place near top of app.py)
def current_prefix():
    now = datetime.now()
    return now.strftime("%y%m")  # YYMM


def next_sequence_for_prefix(prefix):
    conn = db()
    cur = conn.cursor()
    cur.execute("SELECT MAX(job_seq) as max_seq FROM jobs WHERE job_prefix = ?", (prefix,))
    row = cur.fetchone()
    conn.close()
    if row["max_seq"]:
        return row["max_seq"] + 1
    return 1


def build_internal_number(prefix, seq):
    return f"{prefix}{str(seq).zfill(3)}"

@app.post("/jobs/new")
def job_create():
    job_prefix = request.form.get("job_prefix", "").strip()
    job_seq_input = request.form.get("job_seq", "").strip()
    client_reference = request.form.get("client_reference", "").strip()

    client_id = request.form.get("client_id") or None
    customer_id = request.form.get("customer_id") or None
    asset_id = request.form.get("asset_id") or None

    job_type = request.form.get("job_type", "Field Call").strip()
    visit_type = request.form.get("visit_type", "New Visit").strip()
    status = request.form.get("status", "New").strip()
    priority = request.form.get("priority", "Normal").strip()
    job_address = request.form.get("job_address", "").strip()
    description = request.form.get("description", "").strip()

    if not job_prefix or len(job_prefix) != 4:
        flash("Prefix must be 4 digits (YYMM).", "danger")
        return redirect(url_for("job_new"))

    if job_seq_input:
        if not job_seq_input.isdigit():
            flash("Sequence must be numeric.", "danger")
            return redirect(url_for("job_new"))
        job_seq = int(job_seq_input)
    else:
        job_seq = next_sequence_for_prefix(job_prefix)

    internal_job_number = build_internal_number(job_prefix, job_seq)

    conn = db()
    cur = conn.cursor()

    # Check for collision
    cur.execute("SELECT id FROM jobs WHERE internal_job_number = ?", (internal_job_number,))
    if cur.fetchone():
        conn.close()
        flash("Job number already exists for that prefix/sequence.", "danger")
        return redirect(url_for("job_new"))

    now = datetime.now().isoformat(timespec="seconds")

    cur.execute("""
        INSERT INTO jobs (
            job_prefix, job_seq, internal_job_number,
            client_reference,
            client_id, customer_id, asset_id,
            job_type, visit_type, status, priority,
            job_address, description,
            created_at, updated_at
        )
        VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
    """, (
        job_prefix, job_seq, internal_job_number,
        client_reference or None,
        client_id, customer_id, asset_id,
        job_type, visit_type, status, priority,
        job_address, description,
        now, now
    ))

    job_id = cur.lastrowid

    cur.execute("""
        INSERT INTO interactions (job_id, event_type, narrative, occurred_at, created_at)
        VALUES (?, ?, ?, ?, ?)
    """, (
        job_id,
        "System",
        f"Job created: {internal_job_number}",
        now,
        now
    ))

    conn.commit()
    conn.close()

    flash("Job created.", "success")
    return redirect(url_for("job_detail", job_id=job_id))
