Database

Add this in init_db():

cur.execute("""
CREATE TABLE IF NOT EXISTS job_documents (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    job_id INTEGER NOT NULL,
    doc_type TEXT NOT NULL,            -- Instructions/PPSR/Contract/Invoice/Authority/Form/Other
    title TEXT,
    original_filename TEXT NOT NULL,
    stored_filename TEXT NOT NULL,     -- what we store in /uploads
    mime_type TEXT,
    file_size INTEGER,
    uploaded_by_user_id INTEGER,
    uploaded_at TEXT NOT NULL,
    notes TEXT,
    FOREIGN KEY(job_id) REFERENCES jobs(id),
    FOREIGN KEY(uploaded_by_user_id) REFERENCES users(id)
)
""")

If you’re already using uploads/ and allowed_file() from earlier, keep that. Update allowed extensions so you can upload common office docs:

ALLOWED_EXTENSIONS = {"png", "jpg", "jpeg", "pdf", "doc", "docx", "xls", "xlsx", "csv"}
Secure file serving

If you don’t already have a protected uploads route, add this (it keeps docs behind login):

from flask import send_from_directory, abort

@app.get("/uploads/<path:filename>")
@login_required
def uploaded_file(filename):
    # NOTE: for production you’d add stronger checks per job access.
    return send_from_directory(app.config["UPLOAD_FOLDER"], filename, as_attachment=False)
Upload route

Add this route to app.py:

import mimetypes
from werkzeug.utils import secure_filename

@app.post("/jobs/<int:job_id>/documents/upload")
@login_required
def job_document_upload(job_id: int):
    doc_type = (request.form.get("doc_type") or "Other").strip()
    title = (request.form.get("title") or "").strip()
    notes = (request.form.get("notes") or "").strip()
    file = request.files.get("file")

    if not file or not file.filename:
        flash("Select a file to upload.", "danger")
        return redirect(url_for("job_detail", job_id=job_id))

    if not allowed_file(file.filename):
        flash("Unsupported file type.", "danger")
        return redirect(url_for("job_detail", job_id=job_id))

    original_filename = secure_filename(file.filename)
    ts = now_ts().replace(":", "").replace("-", "")
    stored_filename = f"job_{job_id}_{ts}_{original_filename}"

    filepath = os.path.join(app.config["UPLOAD_FOLDER"], stored_filename)
    file.save(filepath)

    mime_type = mimetypes.guess_type(original_filename)[0] or "application/octet-stream"
    file_size = os.path.getsize(filepath)

    conn = db()
    cur = conn.cursor()
    cur.execute("""
        INSERT INTO job_documents
        (job_id, doc_type, title, original_filename, stored_filename, mime_type, file_size, uploaded_by_user_id, uploaded_at, notes)
        VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
    """, (job_id, doc_type, title or None, original_filename, stored_filename, mime_type, file_size,
          session.get("user_id"), now_ts(), notes or None))
    doc_id = cur.lastrowid
    conn.commit()
    conn.close()

    try:
        audit("job_document", doc_id, "create", "Job document uploaded",
              {"job_id": job_id, "doc_type": doc_type, "filename": original_filename})
    except Exception:
        pass

    flash("Document uploaded.", "success")
    return redirect(url_for("job_detail", job_id=job_id))
Fetch documents in job detail

In your job_detail() route, after you load job, add:

cur.execute("""
    SELECT d.*, u.full_name AS uploaded_by
    FROM job_documents d
    LEFT JOIN users u ON u.id = d.uploaded_by_user_id
    WHERE d.job_id = ?
    ORDER BY d.id DESC
""", (job_id,))
documents = cur.fetchall()

Then pass documents=documents into render_template(...).
