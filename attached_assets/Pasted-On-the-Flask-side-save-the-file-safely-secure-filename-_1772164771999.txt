On the Flask side, save the file safely (secure filename, whitelist extensions), store the stored path/filename against the interaction, and optionally generate a thumbnail later. This example stores files under static/uploads/interactions/ so they can be served easily.

import os
from werkzeug.utils import secure_filename
from flask import current_app, request, redirect, url_for, flash
from datetime import datetime

ALLOWED_EXTENSIONS = {"png", "jpg", "jpeg", "webp", "heic"}

def allowed_file(filename: str) -> bool:
    return "." in filename and filename.rsplit(".", 1)[1].lower() in ALLOWED_EXTENSIONS

def parse_interaction_datetime(date_str, time_str):
    combined = f"{date_str} {time_str}".upper().strip()
    return datetime.strptime(combined, "%d/%m/%Y %I:%M %p")

@bp.post("/jobs/<int:job_id>/interactions")
def add_interaction(job_id):
    job = Job.query.get_or_404(job_id)

    interaction_type = request.form.get("interaction_type", "").strip()
    narrative = request.form.get("narrative", "").strip()
    date_str = request.form.get("interaction_date", "").strip()
    time_str = request.form.get("interaction_time", "").strip()

    occurred_at = None
    if date_str and time_str:
        occurred_at = parse_interaction_datetime(date_str, time_str)

    photo = request.files.get("attendance_photo")
    photo_path = None

    if photo and photo.filename:
        filename = secure_filename(photo.filename)
        if not allowed_file(filename):
            flash("Unsupported image type.", "error")
            return redirect(url_for("jobs.view", job_id=job_id))

        upload_dir = os.path.join(current_app.root_path, "static", "uploads", "interactions", str(job_id))
        os.makedirs(upload_dir, exist_ok=True)

        # Optional: prefix to avoid collisions
        stored_name = f"{int(datetime.utcnow().timestamp())}_{filename}"
        full_path = os.path.join(upload_dir, stored_name)
        photo.save(full_path)

        photo_path = f"uploads/interactions/{job_id}/{stored_name}"

    interaction = Interaction(
        job_id=job_id,
        interaction_type=interaction_type,
        narrative=narrative,
        occurred_at=occurred_at,

Auto status change when an admin adds a field note

You’ll need two things: a way to determine “admin” (role/flag on the user), and a rule for what status to set. The clean approach is to only trigger this when the interaction type is “Field Note” (or whatever you call it) and the user is admin.

Here’s a simple helper that updates the Job status:

def maybe_apply_admin_status_change(job, interaction):
    # Adjust these to match your system
    is_admin = getattr(current_user, "is_admin", False) or ("admin" in getattr(current_user, "roles", []))

    if not is_admin:
        return

    # Trigger condition: admin adds a field note
    if interaction.interaction_type.lower() in {"field note", "fieldnote", "note"}:
        # Choose your desired status value
        job.status = "IN_PROGRESS"  # or "ACTIVE", "ACTIONED", etc.

If you want the status to depend on the note category (e.g., “Hold”, “Escalated”, “Awaiting Client”), you can add a dropdown on the “Add Interaction” form and map it to statuses, but the above gets the automatic behaviour immediately.
